Mamba

Mamba cheat sheet

January 6, 2024

  • Article

Installation

For Linux, macOS, or Git Bash on Windows install with:

"${SHELL}" <(curl -L micro.mamba.pm/install.sh)

Uninstallation

# Remove the micromamba init lines
micromamba shell deinit
# Remove related files
rm -rf ~/micromamba ~/.mamba ~/.mambarc ~/.condarc ~/.local/bin/micromamba

Self-update

micromamba config append channels conda-forge
micromamba config set channel_priority strict
micromamba self-update

Shell initialization

eval "$(micromamba shell hook --shell bash)"
micromamba activate

Environment

micromamba create -n <env>                    # Create env without any packages
micromamba create -n <env> <list of packages> # Create env with packages
micromamba create -n <env> -f <file>.txt      # Create env from .txt file
micromamba create -n <env> -f <file>.yml      # Create env from .yml file
micromamba activate                           # Activate base
micromamba activate base                      # Activate base
micromamba activate <env>                     # Activate env
micromamba deactivate                         # Deactivate current env
micromamba remove <env>                       # Delete env
micromamba env list                           # List all env
micromamba env remove                         # Remove current env

Package

micromamba install python=3.10         # Install Python
micromamba install <list of packages>  # Install packages
micromamba remove <list of packages>   # Remove packages
micromamba remove -a                   # Remove all packages in active env
micromamba update <list of packages>   # Update packages
micromamba clean                       # Clean package cache
micromamba list                        # List all packages in active env
micromamba list <package>              # List specific package in active env
micromamba search <package>            # Find specific package in active env or channels
